Automated Asset Bundling

I developed a custom Blender Addon for my team to standardize naming conventions and automate the EXPORT/ORPHAN collection hierarchy.

  • Zero Manual Parenting: Script auto-detects asset types and parents them to logical "Empties" based on dynamic group data.
  • Material Auto-Linking: Integrates with a central .blend library to automatically assign production-ready materials during the renaming phase.
  • Workflow Speed: Introduced a custom Pie Menu system, reducing naming/organization time by approximately 60%.
  • Parametric Generation: Companion script morphs 2 base meshes across 25 real-world dimensions, producing 275 asset variants without manual duplication.

def bundle_selected_objects():
    selected_objs = [obj for obj in bpy.context.selected_objects if obj.type in {'MESH', 'CURVE'}]

    export_col = ensure_collection("EXPORT", color_tag='COLOR_04')
    placeholder = ensure_empty("PLACEHOLDER")

    for obj in selected_objs:
        name = obj.name.lower()

        for group, subs in GROUPS.items():
            if any(sub.lower() in name for sub in subs):
                # Automated Hierarchy Logic
                if group not in group_empties:
                    group_empties[group] = ensure_empty(group)
                    safe_link_to_collection(group_empties[group], export_col)
                    group_empties[group].parent = placeholder

                safe_link_to_collection(obj, export_col)
                obj.parent = group_empties[group]

    # Deep copy mesh data — prevents unintended object linking
    left_obj_copy.data  = left_obj.data.copy()
    right_obj_copy.data = right_obj.data.copy()

    if left_obj_copy.data.shape_keys:
        sk = left_obj_copy.data.shape_keys.key_blocks
        sk["_TOP_B_(width)"].value = width_key_value
        sk["_TOP_B_(depth)"].value = depth_key_value

    left_obj_copy.name  = f"{collection_name}_TOP_B_base_LEFT_{y_dim}x{x_dim}"
    right_obj_copy.name = f"{collection_name}_TOP_B_base_RIGHT_{y_dim}x{x_dim}"
